#include <Arduino.h>
#include <TM1637Display.h>

// Define the connections pins
#define CLK D1
#define DIO D2
TM1637Display display(CLK, DIO); // Create an instance of the TM1637Display

uint8_t data[] = {
      SEG_A | SEG_F | SEG_G | SEG_E | SEG_D | SEG_DP,  // E
      SEG_A | SEG_F | SEG_G | SEG_C | SEG_D | SEG_DP,  // S
      };

uint8_t dots[] = {
      SEG_DP
};

long int timer = 0;
int interval = 300;
int seconds = 0;
int minutes = 0;
int Time[] = {0, 0};

void setup() {
  display.setBrightness(5);  // Set the display brightness (0-7)

};


void loop() {
  
  if (millis() - timer >= interval){
    timer += interval;
    seconds++; //1sekunde vergangen
  


    if (seconds == 60){ // nach 60sek +1min 60sek reset
      seconds = 0;
      minutes++;
    }

      ////// Display String 

    display.clear();  // Clear the display

    if (minutes == 0){
      display.showNumberDec(seconds, true, 2, 2);
      
    }
    else{
      display.setSegments(dots, 1, 1);        //doppelpunkt zeigen
      display.showNumberDec(minutes, true, 2, 0);   //minuten auf den ersten beiden
      display.showNumberDec(seconds, true, 2, 2);   //sekunden auf den letzten beiden
    }
    


    
  }



  ////// Display Number

};
